----------------------------------------------------------------
  ONE-TIME SETUP
----------------------------------------------------------------

# 1. Clone the repository
git clone <your-repo-url>
cd code-review-assistant

# 2. Install Poetry (if not already installed)
pip install poetry

# 3. Setup backend
cd backend
poetry install

# 4. Install PDF generation dependency
poetry add reportlab

# 5. Install static analysis tools
poetry add bandit semgrep

# 6. Setup frontend
cd ../frontend
npm install

# 7. Create .env file in backend/
cd ../backend
copy .env.example .env       # Windows
cp .env.example .env         # Linux/Mac

# 8. Add your API keys to .env
# GOOGLE_API_KEY=your_gemini_key_here
# OPENAI_API_KEY=your_openai_key_here  (if using OpenAI)


----------------------------------------------------------------
  OLLAMA COMMANDS (Local LLM)
----------------------------------------------------------------

# Install Ollama (Windows)
# Download from: https://ollama.com/download/windows

# Check Ollama version
ollama --version

# Pull a model (one-time download)
ollama pull llama3
ollama pull llama3.1
ollama pull mistral
ollama pull codellama          # Best for code review tasks
ollama pull deepseek-coder     # Alternative for code tasks
ollama pull gemma3

# List all downloaded models
ollama list

# Run a model interactively (test it)
ollama run codellama
ollama run llama3

# Start Ollama server (if not running as service)
ollama serve

# Check if Ollama server is running
curl http://localhost:11434/api/tags

# Remove a model
ollama rm llama3

# Show model info
ollama show codellama

# Pull a specific model size
ollama pull llama3:8b
ollama pull llama3:70b
ollama pull codellama:13b

# Ollama API base URL (use in .env)
# OLLAMA_BASE_URL=http://localhost:11434

# Check Ollama running models
curl http://localhost:11434/api/ps

# Stop a running model
curl -X DELETE http://localhost:11434/api/cancel


----------------------------------------------------------------
  RUNNING THE PROJECT
----------------------------------------------------------------

# Terminal 1 — Start Backend (from backend/)
cd backend
poetry run uvicorn main:app --reload --port 8000

# Terminal 2 — Start Frontend (from frontend/)
cd frontend
npm run dev

# Open in browser
http://localhost:3000

# Backend API docs (Swagger UI)
http://localhost:8000/docs


----------------------------------------------------------------
  RUNNING TESTS

----------------------------------------------------------------
  API ENDPOINTS
----------------------------------------------------------------

POST   /analyze                        Upload file and start analysis
GET    /status/{session_id}            Poll pipeline progress
GET    /report/{session_id}/json       Get JSON report
GET    /report/{session_id}/markdown   Get Markdown report
GET    /report/{session_id}/pdf        Download PDF report
GET    /report/{session_id}/sarif      Get SARIF report
DELETE /session/{session_id}           Cleanup temp files
GET    /                               Health check

----------------------------------------------------------------
  TROUBLESHOOTING
----------------------------------------------------------------

# ModuleNotFoundError in tests
# → Add to top of test file:
#   import sys, os
#   sys.path.insert(0, os.path.abspath(os.path.join(os.path.dirname(__file__), "..")))

# Port 8000 already in use
poetry run uvicorn main:app --reload --port 8001

# Port 3000 already in use
npm run dev -- --port 3001

# Clear temp uploads manually
cd backend
rmdir /s /q temp_uploads        # Windows
rm -rf temp_uploads             # Linux/Mac
mkdir temp_uploads

# Clear all reports manually
cd backend
rmdir /s /q reports             # Windows
rm -rf reports                  # Linux/Mac
mkdir reports

# Reinstall all backend dependencies
poetry install --no-cache

# Reinstall all frontend dependencies
cd frontend
rm -rf node_modules
npm install

# Check if uvicorn is running
curl http://localhost:8000/
